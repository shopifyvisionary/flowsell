// ── Claude API ────────────────────────────────────────────────────────────────
async function callClaude(system, userPrompt) {
  try {
    const res = await fetch("https://api.anthropic.com/v1/messages", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        model: MODEL, max_tokens: 1000, system,
        messages: [{ role: "user", content: userPrompt }],
      }),
    });
    const data = await res.json();
    return data.content?.map(b => b.text || "").join("") || "";
  } catch { return ""; }
}

function parseJSON(raw) {
  try { return JSON.parse(raw.replace(/```json|```/g, "").trim()); }
  catch { return null; }
}

// ── Reusable UI ───────────────────────────────────────────────────────────────
function GoldBtn({ children, onClick, disabled, full, outline, small, style = {} }) {
  return (
    <button onClick={onClick} disabled={disabled} style={{
      width: full ? "100%" : "auto",
      padding: small ? "8px 16px" : "13px 24px",
      background: outline ? "transparent" : disabled ? "#333" : `linear-gradient(135deg, ${C.goldLight}, ${C.gold})`,
      color: outline ? C.gold : disabled ? C.textDim : "#000",
      border: outline ? `1.5px solid ${C.gold}` : "none",
      borderRadius: 10, fontFamily: "'DM Sans', sans-serif",
      fontWeight: 700, fontSize: small ? 12 : 13,
      cursor: disabled ? "not-allowed" : "pointer",
      opacity: disabled ? 0.6 : 1,
      transition: "all 0.2s", letterSpacing: "0.02em",
      ...style
    }}
      onMouseEnter={e => { if (!disabled && !outline) e.currentTarget.style.filter = "brightness(1.1)"; }}
      onMouseLeave={e => { e.currentTarget.style.filter = ""; }}
    >{children}</button>
  );
}

// ── ROOT ──────────────────────────────────────────────────────────────────────
export default function FlowSell() {
  const [screen, setScreen] = useState("landing"); // landing | plan | login | app
  const [selectedPlan, setSelectedPlan] = useState(null);
  const [user, setUser] = useState(null);

  if (screen === "landing") return (
    <Landing
      onLogin={() => setScreen("login")}
      onPlan={plan => { setSelectedPlan(plan); setScreen("plan"); }}
    />
  );

  if (screen === "plan") return (
    <PlanSelect
      plan={selectedPlan}
      onBack={() => setScreen("landing")}
      onConfirm={plan => { setSelectedPlan(plan); setScreen("login"); }}
    />
  );

  if (screen === "login") return (
    <Login
      onBack={() => setScreen(selectedPlan ? "plan" : "landing")}
      onLogin={u => { setUser({ ...u, plan: selectedPlan || PLANS[1] }); setScreen("app"); }}
    />
  );

  if (screen === "app" && user) return (
    <Dashboard user={user} onLogout={() => { setUser(null); setScreen("landing"); }} />
  );

  return null;
}
